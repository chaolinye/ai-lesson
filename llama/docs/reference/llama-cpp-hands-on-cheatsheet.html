下载模型

推荐测试模型（小、快）：

# 方法 1：直接用 -hf 下载并启动（一次性）
llama-server -hf Qwen/Qwen2.5-1.5B-Instruct-GGUF:Q4_K_M

# 方法 2：先下载模型文件到本地
# 从 Hugging Face 手动下载 GGUF 文件到 C:\models\

启动 server

llama-server ^
  -m C:\models\model.Q5_K_M.gguf ^
  --host 127.0.0.1 --port 8080 ^
  -ngl all -c 8192 -fa on ^
  -np 1 --temp 0.7 ^
  --api-key "sk-llama-local-key"

测试 API

# Chat completion
curl http://127.0.0.1:8080/v1/chat/completions ^
  -H "Content-Type: application/json" ^
  -H "Authorization: Bearer sk-llama-local-key" ^
  -d "{\"model\":\"local\",\"messages\":[{\"role\":\"user\",\"content\":\"你好\"}],\"temperature\":0.7}"

# Streaming
curl http://127.0.0.1:8080/v1/chat/completions ^
  -H "Content-Type: application/json" ^
  -H "Authorization: Bearer sk-llama-local-key" ^
  -d "{\"model\":\"local\",\"messages\":[{\"role\":\"user\",\"content\":\"数到3\"}],\"stream\":true}"

# JSON schema 约束
curl http://127.0.0.1:8080/v1/chat/completions ^
  -H "Content-Type: application/json" ^
  -H "Authorization: Bearer sk-llama-local-key" ^
  -d "{\"model\":\"local\",\"messages\":[{\"role\":\"user\",\"content\":\"列出三种编程语言\"}],\"response_format\":{\"type\":\"json_object\",\"schema\":{\"type\":\"object\",\"properties\":{\"languages\":{\"type\":\"array\",\"items\":{\"type\":\"string\"}}}}}}"
